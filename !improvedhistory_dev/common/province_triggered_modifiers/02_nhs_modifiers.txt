nhs_inward_focus_build_cost_reduction = {
	potential = {
		always = yes
	}
	
	#viewer = {
	#}
	
	trigger = {
		owner = {
			custom_trigger_tooltip = {
				tooltip = nhs_completed_overlord_inward_focus_comp_tt
				OR = {
					has_country_flag = nhs_completed_inward_focus
					overlord = {
						has_country_flag = nhs_completed_inward_focus
