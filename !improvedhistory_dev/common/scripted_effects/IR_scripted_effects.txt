control_mission_cede = {
	every_province = {
		limit = { $PROVINCES$ = yes NOT = { country_or_non_sovereign_subject_holds = ROOT } country_or_non_sovereign_subject_controls = { TAG=ROOT } }
		hidden_effect = { 
			if = {
				limit = { owner = { war_with = ROOT } }
				white_peace = owner 
			}
			change_controller = ROOT
		}
		cede_province = ROOT
	}
}

discovery_mission_cede = {
	if = { 
		limit = { $PROVINCE_ID$ = { is_empty = yes } NOT = { has_global_flag = IR_discovery_missions_disabled } }
		hidden_effect = {
			$PROVINCE_ID$ = {
				change_trade_goods = $TRADE_GOOD$
				change_religion = ROOT	
				change_culture = ROOT
				add_territorial_core = ROOT	
			}
		}
		add_treasury = -$COST$
		$PROVINCE_ID$ = { 
			cede_province = ROOT 
			add_province_modifier = {
				name = $MODIFIER$
				duration = $DURATION$
			}
		}
	}
	else_if = { 
		limit = { $PROVINCE_ID$ = { is_city = yes NOT = { country_or_non_sovereign_subject_holds = ROOT } } NOT = { has_global_flag = IR_discovery_missions_disabled } }
		if = {
			limit = { 
				$PROVINCE_ID$ = { owner = { is_colonizer_tech_group = no } }
			}
			hidden_effect = {
				$PROVINCE_ID$ = {
					owner = { country_event = { id = orion_seize.1 days = 0 } }
					change_trade_goods = $TRADE_GOOD$
					add_territorial_core = ROOT	
				}	
			}
			add_treasury = -$COST$
			$PROVINCE_ID$ = { 
				cede_province = ROOT 
				add_province_modifier = {
					name = $MODIFIER$
					duration = $DURATION$
				}
			}
		}
		else = { #changed from #else_if
		#	limit = { $PROVINCE_ID$ = { owner = { is_colonizer_tech_group = yes } } }
			$PROVINCE_ID$ = { add_permanent_claim = ROOT }
		}
	}
}

	}	
}

colony_mission_cede = {
	add_treasury = -$COST$
	custom_tooltip = $TOOLTIP$
	every_province = {
		limit = { $PROVINCES$ = yes NOT = { owned_by = ROOT } }
		if = { 
			limit = { is_empty = yes  NOT = { has_global_flag = IR_colony_missions_disabled } }
			hidden_effect = {
				change_religion = ROOT	
				change_culture = ROOT
				add_territorial_core = ROOT	
				cede_province = ROOT
			}
		}
		else_if = { 
			limit = { is_city = yes NOT = { country_or_non_sovereign_subject_holds = ROOT } NOT = { has_global_flag = IR_colony_missions_disabled } }
			if = {
				limit = { owner = { is_colonizer_tech_group = no } }
				hidden_effect = {
					owner = { country_event = { id = orion_seize.1 days = 0 } }
					add_territorial_core = ROOT		
					cede_province = ROOT 
				}
			}
			else = { add_permanent_claim = ROOT }
		}
	}
}

custom_subject = {
	custom_tooltip = $TOOLTIP$
	hidden_effect = {
		if = {
			limit = { $TAG$ = { exists = no } }
			every_province = {
				limit = { $PROVINCES$ = yes }
				if = { 
					limit = { NOT = { owned_by = ROOT } }
					add_core = $TAG$
					cede_province = $TAG$
				}
				else_if = { 
					limit = { owned_by = ROOT }
					add_core = $TAG$
					cede_province = $TAG$
				}	
			}
			$TAG$ = {
				
				define_ruler = {
 					name = "$LEADER$"
   					dynasty = "$DYNASTY$"
    				adm = $ADM$
    				dip = $DIP$
    				mil = $MIL$
    				age = $AGE$
    				claim = $CLAIM$
    				culture = $CULTURE$
    				religion = $RELIGION$
    				fixed = $FIXED$
    				$GENDER$ = yes
				}
			}
			create_subject = {
				subject_type = $SUBJECT_TYPE$
				subject = $TAG$
			}
		}
	}
}
