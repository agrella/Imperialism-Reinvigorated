# Aidanon Gilbertakos
country_event = {
	id = nhs_tribute.1
	title = newnhs_tributeevents.t
	desc = newnhs_tributeevents.d
	picture = UNIVERSITY_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		nhs_condition_all_ely = { KOMANDA = tag }
		is_at_war = no
		NOT = { has_country_modifier = nhs_handicap_settlers }
	}
	
	
	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.75
			stability = 1
		}
		modifier = {
			factor = 0.75
			stability = 2
		}
		modifier = {
			factor = 0.75
			stability = 3
		}
		modifier = {
			factor = 0.75
			any_owned_province = {
				has_province_modifier = nhs_gilded_keep
			}
		}
	}
	
	option = {
		name = newnhs_tributeevents.a
		#set_country_flag = newnhs_great_advisor
		if = {
			limit = {
				NOT = { monthly_income = 25 }
			}
		
			define_advisor = {
				type = aidanon_gilbertakos
				name = "Aidanon Gilbertakos"
				skill = 1
				discount = yes
			}
		}
		
		if = {
			limit = {
				monthly_income = 25
				NOT = { monthly_income = 50 }
			}
		
			define_advisor = {
				type = aidanon_gilbertakos
				name = "Aidanon Gilbertakos"
				skill = 2
				discount = yes
			}
		}
		
		if = {
			limit = {
				monthly_income = 50
			}
		
			define_advisor = {
				type = aidanon_gilbertakos
				name = "Aidanon Gilbertakos"
				skill = 3
				discount = yes
			}
		}
		add_dip_power = 25
		add_mil_power = 25
		add_adm_power = 25
	}
}
